```rust
use ringbuf::{HeapRb, traits::Split};
use ringbuf::HeapProd;
use ringbuf::HeapCons;
use tokio::sync::mpsc;

use super::dispatch::ShardMessage;

/// Default SPSC buffer size (entries, not bytes). 4096 entries.
pub const CHANNEL_BUFFER_SIZE: usize = 4096;

/// Connection channel capacity for listener -> shard.
pub const CONN_CHANNEL_CAPACITY: usize = 256;

/// SPSC channel mesh connecting all shard pairs.
///
/// For N shards, creates N*(N-1) unidirectional SPSC channels.
/// `producers[i][j]` sends from shard i to shard j (j < i maps differently due to skip-self).
/// Access via `senders_for(shard_id)` and `receivers_for(shard_id)`.
pub struct ChannelMesh {
    /// producers[sender_shard] = Vec of producers, one per other shard.
    /// Index within the Vec corresponds to target shard, but skipping self.
    producers: Vec<Vec<HeapProd<'static, ShardMessage>>>,
    /// consumers[receiver_shard] = Vec of consumers, one per other shard.
    consumers: Vec<Vec<HeapCons<'static, ShardMessage>>>,
    /// Connection channels: listener sends TcpStream to each shard.
    conn_txs: Vec<mpsc::Sender<tokio::net::TcpStream>>,
    conn_rxs: Vec<Option<mpsc::Receiver<tokio::net::TcpStream>>>,
    num_shards: usize,
}

impl ChannelMesh {
    /// Create a new channel mesh for `n` shards.
    ///
    /// Each shard pair gets one SPSC channel in each direction.
    /// Connection channels use tokio mpsc (single listener -> each shard).
    pub fn new(n: usize, buffer_size: usize) -> Self {
        // For simplicity, use a flat mapping: producers[i] has entries for shards 0..n excluding i.
        // The index mapping: for shard i sending to shard j:
        //   if j < i: index = j
        //   if j > i: index = j - 1
        let mut producers: Vec<Vec<HeapProd<'static, ShardMessage>>> = Vec::with_capacity(n);
        let mut consumers: Vec<Vec<HeapCons<'static, ShardMessage>>> = Vec::with_capacity(n);

        // Initialize empty vecs
        for _ in 0..n {
            producers.push(Vec::with_capacity(n - 1));
            consumers.push(Vec::with_capacity(n - 1));
        }

        // Create channels: for each ordered pair (sender, receiver) where sender != receiver
        for sender in 0..n {
            for receiver in 0..n {
                if sender == receiver { continue; }
                let rb = HeapRb::new(buffer_size);
                let (prod, cons) = rb.split();
                // SAFETY: we leak the ring buffer so producers/consumers have 'static lifetime.
                // This is intentional -- the mesh lives for the entire server lifetime.
                let prod: HeapProd<'static, ShardMessage> = unsafe { std::mem::transmute(prod) };
                let cons: HeapCons<'static, ShardMessage> = unsafe { std::mem::transmute(cons) };
                producers[sender].push(prod);
                consumers[receiver].push(cons);
            }
        }

        // Connection channels
        let mut conn_txs = Vec::with_capacity(n);
        let mut conn_rxs = Vec::with_capacity(n);
        for _ in 0..n {
            let (tx, rx) = mpsc::channel(CONN_CHANNEL_CAPACITY);
            conn_txs.push(tx);
            conn_rxs.push(Some(rx));
        }

        ChannelMesh {
            producers,
            consumers,
            conn_txs,
            conn_rxs,
            num_shards: n,
        }
    }

    /// Take the producers for a specific shard (call once during setup).
    /// Returns Vec of producers indexed by target shard (skip-self mapping).
    pub fn take_producers(&mut self, shard_id: usize) -> Vec<HeapProd<'static, ShardMessage>> {
        std::mem::take(&mut self.producers[shard_id])
    }

    /// Take the consumers for a specific shard (call once during setup).
    pub fn take_consumers(&mut self, shard_id: usize) -> Vec<HeapCons<'static, ShardMessage>> {
        std::mem::take(&mut self.consumers[shard_id])
    }

    /// Take the connection receiver for a specific shard (call once during setup).
    pub fn take_conn_rx(&mut self, shard_id: usize) -> mpsc::Receiver<tokio::net::TcpStream> {
        self.conn_rxs[shard_id].take().expect("conn_rx already taken")
    }

    /// Get a clone of the connection sender for a specific shard (listener uses this).
    pub fn conn_tx(&self, shard_id: usize) -> mpsc::Sender<tokio::net::TcpStream> {
        self.conn_txs[shard_id].clone()
    }

    /// Number of shards in this mesh.
    pub fn num_shards(&self) -> usize {
        self.num_shards
    }

    /// Map a target shard ID to the index within a shard's producer/consumer Vec.
    /// For shard `my_id` sending to `target_id`:
    ///   if target < my_id: index = target
    ///   if target > my_id: index = target - 1
    #[inline]
    pub fn target_index(my_id: usize, target_id: usize) -> usize {
        if target_id < my_id { target_id } else { target_id - 1 }
    }
}
```

NOTE: The `transmute` for `'static` lifetime is needed because `ringbuf::split()` returns references tied to the `HeapRb` lifetime. Since the mesh lives for the entire server process, this is safe. However, if the executor finds a cleaner approach (e.g., using `Box::leak` or `ringbuf`'s owned split API), prefer that. Check `ringbuf` 0.4 docs for `HeapRb::new(n).split()` return types -- if they already return owned `HeapProd`/`HeapCons`, remove the transmute entirely.

IMPORTANT: Check the actual `ringbuf` 0.4.8 API. The `HeapRb::new(n)` returns a `HeapRb<T>`, and `.split()` from the `Split` trait may return owned types. If so, no transmute needed -- just use them directly. The key requirement: producers and consumers must be movable to different threads (Send).
