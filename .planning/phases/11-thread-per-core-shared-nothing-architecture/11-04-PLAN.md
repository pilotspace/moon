BETTER APPROACH: Since Tokio current_thread is cooperative, connections cannot truly run concurrently within a shard. Options:

Choose (b) -- `Rc<RefCell<Vec<Database>>>` -- because:
- Single-threaded runtime = no concurrent access
- Zero overhead for local commands (no channel round-trip)
- Direct `borrow_mut()` on the RefCell for write, `borrow()` for read
- Cross-shard dispatch still goes through SPSC

/// Extract the primary key from a parsed command frame for routing.
/// Returns None for keyless commands (PING, DBSIZE, etc).
fn extract_primary_key(cmd: &[u8], args: &[Frame]) -> Option<&Bytes> {
    // Most commands: first arg is the key
    // Exceptions: MGET/MSET/DEL with multiple keys -> handled by coordinator
    // SELECT, PING, ECHO, INFO, etc -> no key, execute locally
    if args.is_empty() { return None; }
    match &args[0] {
        Frame::BulkString(key) => Some(key),
        _ => None,
    }
}

/// Check if a command is a multi-key command requiring VLL coordination.
fn is_multi_key_command(cmd: &[u8]) -> bool {
    cmd.eq_ignore_ascii_case(b"MGET")
        || cmd.eq_ignore_ascii_case(b"MSET")
        || (cmd.eq_ignore_ascii_case(b"DEL") /* args.len() > 1 check at call site */)
        || cmd.eq_ignore_ascii_case(b"UNLINK")
        || cmd.eq_ignore_ascii_case(b"EXISTS")
}

1. Create `src/shard/coordinator.rs`:

```rust
use std::collections::BTreeMap;
use bytes::Bytes;
use crate::protocol::Frame;
use crate::shard::dispatch::{key_to_shard, ShardMessage};

/// Coordinate a multi-key command across shards using VLL pattern.
///
/// Groups keys by target shard, dispatches to each shard in ascending
/// shard-ID order (deadlock prevention), collects results, and assembles
/// the final response.
///
/// `local_execute` handles keys on the local shard without SPSC overhead.
pub async fn coordinate_mget(
    keys: Vec<Bytes>,
    my_shard: usize,
    num_shards: usize,
    db_index: usize,
    local_db: &mut Database,
    dispatch_tx: &mut [HeapProd<'static, ShardMessage>],
) -> Frame {
    // 1. Group keys by shard
    let mut shard_groups: BTreeMap<usize, Vec<(usize, Bytes)>> = BTreeMap::new();
    for (original_idx, key) in keys.iter().enumerate() {
        let target = key_to_shard(key, num_shards);
        shard_groups.entry(target).or_default().push((original_idx, key.clone()));
    }

    // 2. If all on local shard, fast path
    if shard_groups.len() == 1 && shard_groups.contains_key(&my_shard) {
        return crate::command::string::mget(local_db, &keys_to_frames(&keys));
    }

    // 3. Dispatch in ascending shard-ID order (VLL)
    let mut results: Vec<(usize, Frame)> = Vec::new();
    let mut pending_rxs: Vec<(Vec<usize>, tokio::sync::oneshot::Receiver<Vec<Frame>>)> = Vec::new();

    for (shard_id, indexed_keys) in &shard_groups {
        let original_indices: Vec<usize> = indexed_keys.iter().map(|(i, _)| *i).collect();
        if *shard_id == my_shard {
            // Local execution
            let key_frames: Vec<Frame> = indexed_keys.iter()
                .map(|(_, k)| Frame::BulkString(k.clone()))
                .collect();
            local_db.refresh_now();
            // Execute MGET locally for this subset
            for (orig_idx, key) in indexed_keys {
                let entry = local_db.get(key);
                let frame = match entry {
                    Some(e) => match e.value.as_bytes() {
                        Some(v) => Frame::BulkString(Bytes::copy_from_slice(v)),
                        None => Frame::Null,
                    },
                    None => Frame::Null,
                };
                results.push((*orig_idx, frame));
            }
        } else {
            // Remote dispatch
            let (tx, rx) = tokio::sync::oneshot::channel();
            let commands: Vec<(Bytes, Frame)> = indexed_keys.iter()
                .map(|(_, k)| {
                    let cmd = Frame::Array(vec![
                        Frame::BulkString(Bytes::from_static(b"GET")),
                        Frame::BulkString(k.clone()),
                    ]);
                    (k.clone(), cmd)
                })
                .collect();
            let msg = ShardMessage::MultiExecute {
                db_index,
                commands,
                reply_tx: tx,
            };
            let target_idx = crate::shard::mesh::ChannelMesh::target_index(my_shard, *shard_id);
            // try_push with spin-retry on full
            let prod = &mut dispatch_tx[target_idx];
            // ringbuf try_push -- if full, yield and retry
            while prod.try_push(msg).is_err() {
                tokio::task::yield_now().await;
                // NOTE: msg consumed on success, need to recreate on failure
                // BETTER: use a loop with the actual value
            }
            pending_rxs.push((original_indices, rx));
        }
    }

    // 4. Await all remote results
    for (indices, rx) in pending_rxs {
        if let Ok(frames) = rx.await {
            for (idx, frame) in indices.into_iter().zip(frames) {
                results.push((idx, frame));
            }
        }
    }

    // 5. Sort by original index and assemble response
    results.sort_by_key(|(idx, _)| *idx);
    Frame::Array(results.into_iter().map(|(_, f)| f).collect())
}
```
