FILE: src/scripting/sandbox.rs

Implement four public functions:

1. setup_sandbox(lua: &Lua) -> mlua::Result<()>
   Purpose: Restrict Lua globals after Lua::new().
   Lua::new() already excludes io, package, debug, require.
   Additional restrictions needed:
   - Restrict os.* to clock() only: create empty table, copy os.clock() into it, replace os global
   - globals.set("package", mlua::Nil)  -- CVE-2022-0543 historical sandbox escape
   - globals.set("load", mlua::Nil)     -- arbitrary code loading
   - globals.set("dofile", mlua::Nil)   -- file-based loading
   - globals.set("loadfile", mlua::Nil) -- file-based loading
   - globals.set("collectgarbage", mlua::Nil) -- GC manipulation

2. install_timeout_hook(lua: &Lua, timeout: Duration) -> mlua::Result<()>
   Purpose: Install a VM instruction hook that errors after wall-clock timeout.
   Implementation:
   - let deadline = Instant::now() + timeout;
   - lua.set_hook(HookTriggers { every_nth_instruction: Some(100_000), ..Default::default() }, move |_lua, _debug| { ... })
   - Hook body: if Instant::now() >= deadline { return Err(mlua::Error::RuntimeError("ERR Lua script timeout".to_string())) }
   IMPORTANT: The hook closure captures deadline by value (move). It fires every 100_000 VM instructions.

3. remove_timeout_hook(lua: &Lua)
   - lua.remove_hook()
   IMPORTANT: Must be called in BOTH success and error paths of run_script() -- see Pitfall 3.

4. register_redis_api(lua: &Lua) -> mlua::Result<()>
   Purpose: Create the redis.* table and register all functions.
   Steps:
   - let redis_table = lua.create_table()?
   - redis_table.set("call", crate::scripting::bridge::make_redis_call_fn(lua, true)?)
   - redis_table.set("pcall", crate::scripting::bridge::make_redis_call_fn(lua, false)?)
   - redis_table.set("log", lua.create_function(|_lua, (level, msg): (mlua::Value, String)| { ... tracing::info! ... })?)
   - redis_table.set("error_reply", lua.create_function(|lua, msg: String| { t = lua.create_table(); t.set("err", msg); Ok(t) })?)
   - redis_table.set("status_reply", lua.create_function(|lua, msg: String| { t = lua.create_table(); t.set("ok", msg); Ok(t) })?)
   - redis_table.set("sha1hex", lua.create_function(|_lua, s: mlua::String| { Ok(sha1_smol::Sha1::from(s.as_bytes()).hexdigest()) })?)
   - redis_table.set("LOG_DEBUG", 0i64)?
   - redis_table.set("LOG_VERBOSE", 1i64)?
   - redis_table.set("LOG_NOTICE", 2i64)?
   - redis_table.set("LOG_WARNING", 3i64)?
   - lua.globals().set("redis", redis_table)?
   Imports needed: sha1_smol, tracing

FILE: src/scripting/types.rs

Implement two pub functions:

1. lua_value_to_frame(lua: &Lua, value: &LuaValue) -> mlua::Result<Frame>
   Redis-compatible conversion (from RESEARCH.md Pattern 5):
   - LuaValue::Nil => Ok(Frame::Null)
   - LuaValue::Boolean(false) => Ok(Frame::Null)
   - LuaValue::Boolean(true) => Ok(Frame::Integer(1))
   - LuaValue::Integer(n) => Ok(Frame::Integer(*n))
   - LuaValue::Number(f) => Ok(Frame::Integer(*f as i64))  // truncate -- Redis behavior (Pitfall 6)
   - LuaValue::String(s) => Ok(Frame::BulkString(Bytes::copy_from_slice(s.as_bytes())))
   - LuaValue::Table(t):
     First check: if t.get::<LuaValue>("ok") is LuaValue::String(s) -> Frame::SimpleString(Bytes::copy_from_slice(s.as_bytes()))
     Second check: if t.get::<LuaValue>("err") is LuaValue::String(e) -> Frame::Error(Bytes::copy_from_slice(e.as_bytes()))
     Otherwise: iterate integer keys 1, 2, 3... stopping at first LuaValue::Nil, recurse lua_value_to_frame for each -> Frame::Array
   - other types: return Err(mlua::Error::RuntimeError("ERR Lua type '...' not supported"))

2. frame_to_lua_value<'lua>(lua: &'lua Lua, frame: &Frame) -> mlua::Result<LuaValue<'lua>>
   Redis-compatible reverse conversion:
   - Frame::Integer(n) => Ok(LuaValue::Integer(*n))
   - Frame::BulkString(b) => Ok(LuaValue::String(lua.create_string(b.as_ref())?))
   - Frame::SimpleString(b) => create table {ok: string}, return Ok(LuaValue::Table(t))
   - Frame::Error(e) => create table {err: string}, return Ok(LuaValue::Table(t))
   - Frame::Null => Ok(LuaValue::Boolean(false))
   - Frame::Array(items) => create table, set(i+1, frame_to_lua_value(item)?) for each, return Ok(LuaValue::Table(t))
   - Frame::Boolean(b) => Ok(LuaValue::Integer(if *b { 1 } else { 0 }))
   - Frame::Double(f) => Ok(LuaValue::Integer(f.into_inner() as i64))
   - _ (other RESP3 variants) => Ok(LuaValue::Boolean(false))

   static CURRENT_DB_IDX: Cell<usize> = const { Cell::new(0) };
   static CURRENT_DB_COUNT: Cell<usize> = const { Cell::new(1) };

pub fn set_script_db(db: &mut crate::storage::Database, db_idx: usize, db_count: usize)
   Sets all four thread-locals. Resets SCRIPT_HAD_WRITE to false.

pub fn clear_script_db()
   Sets CURRENT_DB to null_mut().

pub fn script_had_write() -> bool
   Returns SCRIPT_HAD_WRITE value.

pub fn make_redis_call_fn(lua: &Lua, propagate_errors: bool) -> mlua::Result<LuaFunction>
   lua.create_function(move |lua, args: LuaMultiValue| { ... })
   Body:
   1. Map args to Vec<Frame> via crate::scripting::types::lua_value_to_frame(lua, v) for each arg
   2. Error if frames is empty: "ERR Please specify at least one argument for redis.call()"
   3. Extract cmd_bytes from frames[0] (BulkString or SimpleString only; error otherwise)
   4. CURRENT_DB.with(|cell| {
        let ptr = cell.get() as *mut crate::storage::Database;
        if ptr.is_null() { return Err(mlua::Error::RuntimeError("ERR No database context".to_string())); }
        let db = unsafe { &mut *ptr };
        let mut db_idx = CURRENT_DB_IDX.with(|c| c.get());
        let db_count = CURRENT_DB_COUNT.with(|c| c.get());
        if crate::persistence::aof::is_write_command(&cmd_bytes) { SCRIPT_HAD_WRITE.with(|c| c.set(true)); }
        let r = crate::command::dispatch(db, &cmd_bytes, &frames[1..], &mut db_idx, db_count);
        Ok(match r { DispatchResult::Response(f) | DispatchResult::Quit(f) => f })
      })?
   5. If propagate_errors && Frame::Error(e): return Err(mlua::Error::RuntimeError(String::from_utf8_lossy(e).to_string()))
   6. crate::scripting::types::frame_to_lua_value(lua, &result)

setup_lua_vm() -> mlua::Result<Rc<Lua>>:
   let lua = Rc::new(Lua::new());
   sandbox::setup_sandbox(&lua)?;
   sandbox::register_redis_api(&lua)?;
   Ok(lua)

validate_keys_same_shard(keys, shard_id, num_shards) -> Option<Frame>:
   For each key: if key_to_shard(key, num_shards) != shard_id, return Some(Frame::Error(b"CROSSSLOT Keys in script don't hash to the same slot and shard"))
   Return None if all keys local.
   Skip validation if num_shards <= 1.

parse_eval_args(args) -> Result<(Bytes, usize, Vec<Bytes>, Vec<Bytes>), Frame>:
   Validate args.len() >= 2; extract script (args[0]), numkeys (args[1] parsed as usize),
   validate args.len() >= 2 + numkeys, extract keys and argv slices.
   (Full code pattern in RESEARCH.md Code Examples section)

handle_eval(lua, cache, args, db, shard_id, num_shards, selected_db, db_count) -> Frame:
   1. parse_eval_args(args) -- return error frame on failure
   2. validate_keys_same_shard if num_shards > 1 -- return CROSSSLOT if failed
   3. cache.borrow_mut().load(script.clone()) -- cache the script
   4. run_script(lua, &script, keys, argv, db, selected_db, db_count)

handle_evalsha(lua, cache, args, db, shard_id, num_shards, selected_db, db_count) -> Frame:
   1. Extract sha1_hex from args[0]; lowercase it
   2. Borrow cache, look up script; return NOSCRIPT error if not found
   3. Clone the script bytes (drop borrow before calling run_script)
   4. Reconstruct eval_args = [BulkString(script)] ++ args[1..]
   5. parse_eval_args to get keys/argv
   6. validate_keys_same_shard if num_shards > 1
   7. run_script(lua, &script, keys, argv, db, selected_db, db_count)
   NOSCRIPT message: b"NOSCRIPT No matching script. Please use EVAL."

handle_script_subcommand(cache, args) -> (Frame, Option<(String, Bytes)>):
   Returns (response_frame, Option<(sha1_hex, script_bytes)>).
   The Option is Some only for SCRIPT LOAD -- signals caller to fan-out script to all shards.
   LOAD: cache.borrow_mut().load(script); return (BulkString(sha), Some((sha, script)))
   EXISTS: map each sha arg through cache.borrow().exists() -> Integer(1 or 0); return (Array(results), None)
   FLUSH: cache.borrow_mut().flush(); return (SimpleString("OK"), None)
   Unknown subcommand: return (Error("ERR unknown subcommand '...' for 'script' command"), None)

run_script(lua, script, keys, argv, db, selected_db, db_count) -> Frame (private fn):
   1. bridge::set_script_db(db, selected_db, db_count)
   2. let timeout = Duration::from_secs(5)
   3. if sandbox::install_timeout_hook(lua, timeout).is_err(): bridge::clear_script_db(); return error frame
   4. let result = (closure that sets KEYS/ARGV globals and calls lua.load(script.as_ref()).eval::<LuaValue>())
      Set globals["KEYS"] = table with 1-indexed key bytes (Pitfall 5: use i+1 not i)
      Set globals["ARGV"] = table with 1-indexed argv bytes
      let val: LuaValue = lua.load(script.as_ref()).eval()?
      types::lua_value_to_frame(lua, &val)
   5. sandbox::remove_timeout_hook(lua)  // ALWAYS -- both success and error paths
   6. bridge::clear_script_db()           // ALWAYS
   7. Match on result:
      Ok(frame) => frame
      Err(mlua::Error::RuntimeError(msg)) if msg.contains("ERR Lua script timeout") =>
        Frame::Error(Bytes::from_static(b"BUSY Lua script timeout exceeded"))
      Err(e) => Frame::Error(Bytes::from(format!("ERR Error running script: {e}")))
