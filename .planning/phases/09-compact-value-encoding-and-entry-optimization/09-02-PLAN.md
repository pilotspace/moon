```rust
use criterion::{criterion_group, criterion_main, Criterion, black_box};
use bytes::Bytes;
use std::collections::HashMap;

// Import the project types
use rust_redis::storage::entry::{Entry, CompactEntry, current_secs};

fn bench_compact_entry_1m_small_strings(c: &mut Criterion) {
    let base_ts = current_secs();
    c.bench_function("1M compact entries (8-byte strings)", |b| {
        b.iter(|| {
            let mut entries: Vec<CompactEntry> = Vec::with_capacity(1_000_000);
            for i in 0u64..1_000_000 {
                let val = format!("{:08}", i);  // 8-byte string, fits in SSO
                let entry = CompactEntry::new_string(Bytes::from(val));
                entries.push(entry);
            }
            black_box(&entries);
            entries
        });
    });
}

fn bench_entry_size_report(c: &mut Criterion) {
    // Not a benchmark per se, but uses Criterion to report sizes
    c.bench_function("entry size comparison", |b| {
        b.iter(|| {
            let compact_size = std::mem::size_of::<CompactEntry>();
            assert_eq!(compact_size, 24);
            // Old Entry was 88 bytes (72 RedisValue + 8 expires_at_ms + 8 metadata)
            let old_size = 88usize;
            let reduction_pct = ((old_size - compact_size) as f64 / old_size as f64) * 100.0;
            assert!(reduction_pct >= 40.0, "Reduction {}% < 40%", reduction_pct);
            black_box(reduction_pct);
            reduction_pct
        });
    });
}

fn bench_hashmap_1m_keys(c: &mut Criterion) {
    let base_ts = current_secs();
    // Measure HashMap<Bytes, CompactEntry> total memory for 1M small string keys
    c.bench_function("HashMap<Bytes,CompactEntry> 1M keys", |b| {
        b.iter(|| {
            let mut map: HashMap<Bytes, CompactEntry> = HashMap::with_capacity(1_000_000);
            for i in 0u64..1_000_000 {
                let key = Bytes::from(format!("key:{:08}", i));
                let val = format!("{:08}", i);
                let entry = CompactEntry::new_string(Bytes::from(val));
                map.insert(key, entry);
            }
            black_box(&map);
            map
        });
    });
}

criterion_group!(benches, bench_compact_entry_1m_small_strings, bench_entry_size_report, bench_hashmap_1m_keys);
criterion_main!(benches);
```
