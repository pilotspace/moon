```rust
/// Linux io_uring event loop for the shard.
///
/// Replaces the Tokio-based event loop with a synchronous io_uring loop:
/// - Uses UringDriver for accept/recv/send (multishot, provided buffers)
/// - Periodic timeout CQEs replace tokio::time::interval for expiry and SPSC drain
/// - Connection handling is synchronous within the event loop (no spawn_local)
///
/// The existing Tokio path (Shard::run) is used on macOS and non-Linux platforms.
#[cfg(target_os = "linux")]
pub fn run_uring(
    &mut self,
    listener_fd: std::os::fd::RawFd,
    mut consumers: Vec<ringbuf::HeapCons<dispatch::ShardMessage>>,
    producers: Vec<ringbuf::HeapProd<dispatch::ShardMessage>>,
    shutdown: tokio_util::sync::CancellationToken,
    aof_tx: Option<tokio::sync::mpsc::Sender<crate::persistence::aof::AofMessage>>,
) {
    use crate::io::{UringDriver, UringConfig, IoEvent, build_get_response_iovecs};
    use crate::protocol;
    use bytes::BytesMut;
    use ringbuf::traits::Consumer;

    info!("Shard {} started (io_uring mode)", self.id);

    let config = UringConfig::default();
    let mut driver = match UringDriver::new(config) {
        Ok(d) => d,
        Err(e) => {
            tracing::error!("Shard {}: failed to create io_uring: {}. Falling back to Tokio.", self.id, e);
            // Fallback: caller should use Shard::run() instead
            return;
        }
    };

    if let Err(e) = driver.init() {
        tracing::error!("Shard {}: failed to init io_uring: {}", self.id, e);
        return;
    }

    // Submit multishot accept on the listener socket
    // In sharded mode, listener distributes connections via conn_rx.
    // For io_uring mode, the shard can optionally accept directly.
    // For now, integrate via the existing SPSC/conn_rx mechanism:
    // The listener still accepts via Tokio and sends TcpStream to shards.
    // The shard's io_uring handles recv/send AFTER the connection is established.
    //
    // This means: connections arrive via conn_rx (TcpStream), shard extracts
    // the raw fd, registers it with UringDriver, and uses io_uring for I/O.
    //
    // Phase 12 optimization: listener multishot accept is handled in listener.rs.

    // Submit initial timeout for periodic tasks
    let _ = driver.submit_timeout_ms(1); // 1ms for SPSC drain
    let mut last_expiry = std::time::Instant::now();
    let expiry_interval = std::time::Duration::from_millis(100);

    // Track per-connection parse state
    let mut conn_parse_bufs: std::collections::HashMap<u32, BytesMut> = std::collections::HashMap::new();

    // Main event loop
    loop {
        if shutdown.is_cancelled() {
            info!("Shard {} shutting down (io_uring)", self.id);
            break;
        }

        match driver.submit_and_wait() {
            Ok(_) => {}
            Err(e) => {
                if e.kind() == std::io::ErrorKind::Interrupted {
                    continue;
                }
                tracing::error!("Shard {}: submit_and_wait error: {}", self.id, e);
                break;
            }
        }

        let events = driver.drain_completions();

        for event in events {
            match event {
                IoEvent::Recv { conn_id, data } => {
                    // Accumulate data in per-connection parse buffer
                    let parse_buf = conn_parse_bufs
                        .entry(conn_id)
                        .or_insert_with(|| BytesMut::with_capacity(4096));
                    parse_buf.extend_from_slice(&data);

                    // Parse and execute all complete frames
                    let parse_config = protocol::ParseConfig::default();
                    loop {
                        match protocol::parse(parse_buf, &parse_config) {
                            Ok(Some(frame)) => {
                                // Execute command against shard databases
                                let response = self.execute_frame(&frame);
                                let mut resp_buf = BytesMut::new();
                                protocol::serialize(&response, &mut resp_buf);
                                let _ = driver.submit_send(
                                    conn_id,
                                    resp_buf.as_ptr(),
                                    resp_buf.len() as u32,
                                );
                                // NOTE: resp_buf must live until send CQE.
                                // In this synchronous loop, we process CQEs before
                                // the next iteration, so the buffer is safe.
                                // For production, use a per-connection write queue.
                                std::mem::forget(resp_buf); // Leak for now -- proper lifecycle in production
                            }
                            Ok(None) => break, // Need more data
                            Err(protocol::ParseError::Incomplete) => break,
                            Err(_) => {
                                // Protocol error: close connection
                                let _ = driver.close_connection(conn_id);
                                conn_parse_bufs.remove(&conn_id);
                                break;
                            }
                        }
                    }
                }
                IoEvent::Disconnect { conn_id } => {
                    let _ = driver.close_connection(conn_id);
                    conn_parse_bufs.remove(&conn_id);
                }
                IoEvent::RecvNeedsRearm { conn_id } => {
                    let _ = driver.rearm_recv(conn_id);
                }
                IoEvent::SendComplete { .. } => {
                    // Send completed, nothing to do
                }
                IoEvent::SendError { conn_id, .. } => {
                    let _ = driver.close_connection(conn_id);
                    conn_parse_bufs.remove(&conn_id);
                }
                IoEvent::Accept { raw_fd } => {
                    // Direct accept (if listener multishot is on this shard)
                    let _ = driver.register_connection(raw_fd);
                }
                IoEvent::AcceptError { .. } => {
                    // Re-submit multishot accept if it was cancelled
                    // (handled by listener, not shard in current architecture)
                }
                IoEvent::Timeout => {
                    // Periodic: drain SPSC channels
                    Self::drain_spsc_direct(&mut self.databases, &mut consumers, &mut self.pubsub_registry);

                    // Periodic: active expiry (every 100ms)
                    if last_expiry.elapsed() >= expiry_interval {
                        for db in self.databases.iter_mut() {
                            crate::server::expiration::expire_cycle_direct(db);
                        }
                        last_expiry = std::time::Instant::now();
                    }

                    // Re-arm timeout
                    let _ = driver.submit_timeout_ms(1);
                }
                IoEvent::Wakeup => {
                    // SPSC wakeup: drain channels
                    Self::drain_spsc_direct(&mut self.databases, &mut consumers, &mut self.pubsub_registry);
                }
            }
        }
    }
}

/// Execute a parsed frame against shard databases (for io_uring path).
/// Simplified version that handles basic command dispatch.
#[cfg(target_os = "linux")]
fn execute_frame(&mut self, frame: &crate::protocol::Frame) -> crate::protocol::Frame {
    use crate::command::{dispatch as cmd_dispatch, DispatchResult};

    let (cmd, args) = match Self::extract_command_static(frame) {
        Some(pair) => pair,
        None => return crate::protocol::Frame::Error(
            bytes::Bytes::from_static(b"ERR invalid command")
        ),
    };

    let db_count = self.databases.len();
    let mut selected = 0usize;
    self.databases[0].refresh_now();
    match cmd_dispatch(&mut self.databases[0], cmd, args, &mut selected, db_count) {
        DispatchResult::Response(f) => f,
        DispatchResult::Quit(f) => f,
    }
}

/// Direct SPSC drain for io_uring path (no Rc<RefCell> wrapper needed).
#[cfg(target_os = "linux")]
fn drain_spsc_direct(
    databases: &mut Vec<Database>,
    consumers: &mut [ringbuf::HeapCons<dispatch::ShardMessage>],
    pubsub_registry: &mut PubSubRegistry,
) {
    use ringbuf::traits::Consumer;
    const MAX_DRAIN: usize = 256;
    let mut drained = 0;

    for consumer in consumers.iter_mut() {
        while drained < MAX_DRAIN {
            match consumer.try_pop() {
                Some(msg) => {
                    drained += 1;
                    // Handle message with direct database access
                    match msg {
                        dispatch::ShardMessage::PubSubFanOut { channel, message } => {
                            pubsub_registry.publish(&channel, &message);
                        }
                        dispatch::ShardMessage::Execute { db_index, command, reply_tx } => {
                            let db_count = databases.len();
                            let db_idx = db_index.min(db_count.saturating_sub(1));
                            databases[db_idx].refresh_now();
                            let (cmd, args) = match Shard::extract_command_static(&command) {
                                Some(pair) => pair,
                                None => {
                                    let _ = reply_tx.send(crate::protocol::Frame::Error(
                                        bytes::Bytes::from_static(b"ERR invalid command format"),
                                    ));
                                    continue;
                                }
                            };
                            let mut selected = db_idx;
                            let result = crate::command::dispatch(
                                &mut databases[db_idx], cmd, args, &mut selected, db_count
                            );
                            let frame = match result {
                                crate::command::DispatchResult::Response(f) => f,
                                crate::command::DispatchResult::Quit(f) => f,
                            };
                            let _ = reply_tx.send(frame);
                        }
                        dispatch::ShardMessage::MultiExecute { db_index, commands, reply_tx } => {
                            let mut results = Vec::with_capacity(commands.len());
                            let db_count = databases.len();
                            let db_idx = db_index.min(db_count.saturating_sub(1));
                            databases[db_idx].refresh_now();
                            for (_key, cmd_frame) in &commands {
                                let (cmd, args) = match Shard::extract_command_static(cmd_frame) {
                                    Some(pair) => pair,
                                    None => {
                                        results.push(crate::protocol::Frame::Error(
                                            bytes::Bytes::from_static(b"ERR invalid command format"),
                                        ));
                                        continue;
                                    }
                                };
                                let mut selected = db_idx;
                                let result = crate::command::dispatch(
                                    &mut databases[db_idx], cmd, args, &mut selected, db_count
                                );
                                let frame = match result {
                                    crate::command::DispatchResult::Response(f) => f,
                                    crate::command::DispatchResult::Quit(f) => f,
                                };
                                results.push(frame);
                            }
                            let _ = reply_tx.send(results);
                        }
                        dispatch::ShardMessage::SnapshotRequest { reply_tx } => {
                            let snapshot: Vec<(Vec<(bytes::Bytes, crate::storage::entry::Entry)>, u32)> =
                                databases.iter().map(|db| {
                                    let base_ts = db.base_timestamp();
                                    let entries: Vec<_> = db.data().iter()
                                        .map(|(k, v)| (k.clone(), v.clone()))
                                        .collect();
                                    (entries, base_ts)
                                }).collect();
                            let _ = reply_tx.send(snapshot);
                        }
                        dispatch::ShardMessage::Shutdown => {
                            info!("Received shutdown via SPSC (io_uring path)");
                        }
                        dispatch::ShardMessage::NewConnection(_) => {}
                    }
                }
                None => break,
            }
        }
        if drained >= MAX_DRAIN { break; }
    }
}
```
