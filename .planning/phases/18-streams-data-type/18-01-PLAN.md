```rust
use bytes::Bytes;
use std::collections::{BTreeMap, HashMap};

/// Stream entry ID: <milliseconds>-<sequence>.
/// Ordered by (ms, seq) for BTreeMap keying.
#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash)]
pub struct StreamId {
    pub ms: u64,
    pub seq: u64,
}

impl StreamId {
    pub const ZERO: StreamId = StreamId { ms: 0, seq: 0 };
    pub const MAX: StreamId = StreamId { ms: u64::MAX, seq: u64::MAX };

    /// Parse from bytes: "ms-seq", "ms" (default_seq used), "-" (ZERO), "+" (MAX).
    pub fn parse(s: &[u8], default_seq: u64) -> Result<Self, &'static str> { ... }

    /// Format as "ms-seq" Bytes.
    pub fn to_bytes(self) -> Bytes {
        Bytes::from(format!("{}-{}", self.ms, self.seq))
    }
}

/// A Redis Stream: append-only log with consumer group support.
#[derive(Debug, Clone)]
pub struct Stream {
    /// Entries ordered by ID. Value is Vec of (field, value) pairs.
    pub entries: BTreeMap<StreamId, Vec<(Bytes, Bytes)>>,
    /// Total entry count (tracks additions and deletions).
    pub length: u64,
    /// Last generated entry ID (for monotonic guarantee).
    pub last_id: StreamId,
    /// Consumer groups keyed by group name.
    pub groups: HashMap<Bytes, ConsumerGroup>,
}

/// Consumer group state.
#[derive(Debug, Clone)]
pub struct ConsumerGroup {
    pub last_delivered_id: StreamId,
    pub pel: BTreeMap<StreamId, PendingEntry>,
    pub consumers: HashMap<Bytes, Consumer>,
}

/// A pending (unacknowledged) entry in the PEL.
#[derive(Debug, Clone)]
pub struct PendingEntry {
    pub consumer: Bytes,
    pub delivery_time: u64,
    pub delivery_count: u64,
}

/// Per-consumer state within a group.
#[derive(Debug, Clone)]
pub struct Consumer {
    pub name: Bytes,
    pub pending: BTreeMap<StreamId, ()>,
    pub seen_time: u64,
}

impl Stream {
    pub fn new() -> Self { ... }

    /// Generate next auto-ID. Uses max(now_ms, last_id.ms) for monotonic guarantee.
    /// Pitfall 1: clock can go backward, always compare with last_id.
    pub fn next_auto_id(&mut self) -> StreamId { ... }

    /// Validate an explicit ID (must be > last_id).
    pub fn validate_explicit_id(&self, id: StreamId) -> Result<StreamId, &'static str> { ... }

    /// Add an entry. Returns the assigned ID.
    pub fn add(&mut self, id: StreamId, fields: Vec<(Bytes, Bytes)>) -> StreamId { ... }

    /// Range query [start..=end] with optional count limit.
    pub fn range(&self, start: StreamId, end: StreamId, count: Option<usize>) -> Vec<(StreamId, &Vec<(Bytes, Bytes)>)> { ... }

    /// Reverse range query.
    pub fn range_rev(&self, start: StreamId, end: StreamId, count: Option<usize>) -> Vec<(StreamId, &Vec<(Bytes, Bytes)>)> { ... }

    /// Trim by MAXLEN. If approximate, only trim when len > maxlen * 1.1.
    pub fn trim_maxlen(&mut self, maxlen: u64, approximate: bool) -> u64 { ... }

    /// Trim by MINID -- remove entries with ID < minid.
    pub fn trim_minid(&mut self, minid: StreamId, approximate: bool) -> u64 { ... }

    /// Delete specific entries by ID. Returns count of actually deleted.
    pub fn delete(&mut self, ids: &[StreamId]) -> u64 { ... }

    /// Estimate memory usage.
    pub fn estimate_memory(&self) -> usize { ... }
}
```

**1. Create `src/command/stream.rs`** with all core stream command handlers:

Use shared helpers following the pattern in `src/command/string.rs`:
- `extract_bytes(frame) -> Option<Bytes>` from `crate::server::connection`
- `fn err_wrong_args(cmd: &str) -> Frame` for arity errors
- All handlers take `(db: &mut Database, args: &[Frame]) -> Frame`

Implement these handlers:

**XADD** `pub fn xadd(db: &mut Database, args: &[Frame]) -> Frame`:
- Parse: `XADD key [NOMKSTREAM] [MAXLEN|MINID [=|~] threshold] id field value [field value ...]`
- `*` for auto-ID, explicit `ms-seq` or `ms-*` (auto-seq) accepted
- NOMKSTREAM: if key doesn't exist, return Null instead of creating
- After add, apply trim if MAXLEN/MINID specified
- Return BulkString of the generated ID
- Pitfall 1: use stream.next_auto_id() which handles clock backward
- Pitfall 8: streams persist even when empty (don't delete key on trim to 0)

**XLEN** `pub fn xlen(db: &mut Database, args: &[Frame]) -> Frame`:
- `XLEN key` -- return Integer(stream.length) or Integer(0) if key missing

**XRANGE** `pub fn xrange(db: &mut Database, args: &[Frame]) -> Frame`:
- `XRANGE key start end [COUNT count]`
- `-` = StreamId::ZERO, `+` = StreamId::MAX, or parse explicit ID
- Pitfall 7: both bounds inclusive
- Return nested array: `[[id, [field, value, ...]], ...]`

**XREVRANGE** `pub fn xrevrange(db: &mut Database, args: &[Frame]) -> Frame`:
- `XREVRANGE key end start [COUNT count]` -- note reversed argument order vs XRANGE
- Same range logic but reversed iteration

**XTRIM** `pub fn xtrim(db: &mut Database, args: &[Frame]) -> Frame`:
- `XTRIM key MAXLEN|MINID [=|~] threshold`
- Return Integer(count of removed entries)

**XDEL** `pub fn xdel(db: &mut Database, args: &[Frame]) -> Frame`:
- `XDEL key id [id ...]`
- Return Integer(count of actually deleted entries)

**XREAD** `pub fn xread(db: &mut Database, args: &[Frame]) -> Frame`:
- `XREAD [COUNT count] STREAMS key [key ...] id [id ...]`
- NOTE: BLOCK option is parsed but ignored in this plan (Plan 02 handles blocking)
- `$` resolves to stream.last_id at call time (Pitfall 2)
- Return nested array: `[[key, [[id, [field, value, ...]], ...]], ...]`
- Return Null if no new entries on any stream

Format stream entries as nested RESP arrays:
```rust
fn format_entry(id: StreamId, fields: &[(Bytes, Bytes)]) -> Frame {
    let mut field_frames = Vec::with_capacity(fields.len() * 2);
    for (f, v) in fields {
        field_frames.push(Frame::BulkString(f.clone()));
        field_frames.push(Frame::BulkString(v.clone()));
    }
    Frame::Array(vec![
        Frame::BulkString(id.to_bytes()),
        Frame::Array(field_frames),
    ])
}
```
