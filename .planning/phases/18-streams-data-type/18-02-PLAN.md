impl Stream {
    /// Create a consumer group. Returns Err if group already exists.
    /// If mkstream is false and stream doesn't exist, caller handles.
    pub fn create_group(&mut self, name: Bytes, last_delivered_id: StreamId) -> Result<(), &'static str> {
        if self.groups.contains_key(&name) {
            return Err("BUSYGROUP Consumer Group name already exists");
        }
        self.groups.insert(name, ConsumerGroup {
            last_delivered_id,
            pel: BTreeMap::new(),
            consumers: HashMap::new(),
        });
        Ok(())
    }

    /// Destroy a consumer group. Returns true if it existed.
    pub fn destroy_group(&mut self, name: &[u8]) -> bool {
        self.groups.remove(name).is_some()
    }

    /// Set the last-delivered-id for a group.
    pub fn set_group_id(&mut self, name: &[u8], id: StreamId) -> Result<(), &'static str> { ... }

    /// Read new entries for a consumer group (> semantics).
    /// Auto-creates consumer. Adds entries to PEL. Updates last_delivered_id.
    /// Pitfall 3: > means new entries; Pitfall 6: consumer auto-creation.
    pub fn read_group_new(
        &mut self, group: &Bytes, consumer: &Bytes, count: Option<usize>, noack: bool,
    ) -> Result<Vec<(StreamId, Vec<(Bytes, Bytes)>)>, &'static str> { ... }

    /// Read pending entries for a consumer (0 or explicit ID semantics).
    /// Does NOT add to PEL, just replays what consumer already has pending.
    /// Pitfall 3: 0/explicit means re-read pending, not new.
    pub fn read_group_pending(
        &mut self, group: &Bytes, consumer: &Bytes, start: StreamId, count: Option<usize>,
    ) -> Result<Vec<(StreamId, Vec<(Bytes, Bytes)>)>, &'static str> { ... }

    /// Acknowledge entries. Returns count of successfully acknowledged.
    /// Pitfall 4: silently skip entries not in PEL.
    pub fn xack(&mut self, group: &Bytes, ids: &[StreamId]) -> Result<u64, &'static str> { ... }

    /// Get pending entries summary or detail.
    pub fn xpending_summary(&self, group: &Bytes) -> Result<Frame, &'static str> { ... }
    pub fn xpending_detail(
        &self, group: &Bytes, start: StreamId, end: StreamId, count: usize, consumer_filter: Option<&Bytes>,
    ) -> Result<Frame, &'static str> { ... }

    /// Claim pending entries for a different consumer.
    pub fn xclaim(
        &mut self, group: &Bytes, consumer: &Bytes, min_idle_time: u64, ids: &[StreamId],
    ) -> Result<Vec<(StreamId, Vec<(Bytes, Bytes)>)>, &'static str> { ... }

    /// Auto-claim idle pending entries SCAN-style.
    /// Returns (next_cursor_id, claimed_entries, deleted_ids).
    pub fn xautoclaim(
        &mut self, group: &Bytes, consumer: &Bytes, min_idle_time: u64,
        start: StreamId, count: usize,
    ) -> Result<(StreamId, Vec<(StreamId, Vec<(Bytes, Bytes)>)>, Vec<StreamId>), &'static str> { ... }
}

**2. Add command handlers to `src/command/stream.rs`:**

**XGROUP** `pub fn xgroup(db: &mut Database, args: &[Frame]) -> Frame`:
- Subcommands: CREATE, DESTROY, SETID, CREATECONSUMER, DELCONSUMER
- `XGROUP CREATE key group id [MKSTREAM]` -- $ resolves to stream.last_id; MKSTREAM creates stream if missing
- `XGROUP DESTROY key group` -- returns Integer(1) on success, Integer(0) if not found
- `XGROUP SETID key group id` -- update last_delivered_id
- Return appropriate error for unknown subcommand

**XREADGROUP** `pub fn xreadgroup(db: &mut Database, args: &[Frame]) -> Frame`:
- `XREADGROUP GROUP group consumer [COUNT count] [NOACK] STREAMS key [key ...] id [id ...]`
- NOTE: BLOCK option parsed but ignored here (Task 2 handles blocking)
- For each stream+id pair:
  - If id is `>`: call stream.read_group_new() -- delivers new entries, adds to PEL
  - If id is `0` or explicit: call stream.read_group_pending() -- re-reads pending entries
- Return `NOGROUP` error if group doesn't exist on any stream
- Return nested array same format as XREAD, or Null if no entries

**XACK** `pub fn xack(db: &mut Database, args: &[Frame]) -> Frame`:
- `XACK key group id [id ...]`
- Return Integer(count of acknowledged)

**XPENDING** `pub fn xpending(db: &mut Database, args: &[Frame]) -> Frame`:
- Summary form: `XPENDING key group` -- returns [count, min_id, max_id, [[consumer, count], ...]]
- Detail form: `XPENDING key group start end count [consumer]`
  - Returns [[id, consumer, idle_ms, delivery_count], ...]

**XCLAIM** `pub fn xclaim(db: &mut Database, args: &[Frame]) -> Frame`:
- `XCLAIM key group consumer min-idle-time id [id ...]`
- Returns array of claimed entries (with field-value data)

**XAUTOCLAIM** `pub fn xautoclaim(db: &mut Database, args: &[Frame]) -> Frame`:
- `XAUTOCLAIM key group consumer min-idle-time start [COUNT count]`
- Returns [next_start_id, [[id, [field, value, ...]], ...], [deleted_ids...]]

**XINFO** `pub fn xinfo(db: &mut Database, args: &[Frame]) -> Frame`:
- `XINFO STREAM key` -- returns stream metadata (length, radix-tree-keys, radix-tree-nodes, last-generated-id, groups, first-entry, last-entry)
- `XINFO GROUPS key` -- returns array of group info (name, consumers, pending, last-delivered-id)
- `XINFO CONSUMERS key group` -- returns array of consumer info (name, pending, idle)
- Return as flat array of key-value pairs (RESP2 format)

**2. Add `try_wake_stream_waiter` to `src/blocking/wakeup.rs`**:
```rust
/// Called after XADD successfully adds an entry to a stream key.
/// Pops the first waiter (FIFO) and checks if the stream has entries > last_seen_id.
/// For XRead: delivers entries from stream.range(last_seen_id+1.., count).
/// For XReadGroup with >: delivers via stream.read_group_new().
pub fn try_wake_stream_waiter(
    registry: &mut BlockingRegistry,
    db: &mut Database,
    db_index: usize,
    key: &Bytes,
) -> bool {
    while registry.has_waiters(db_index, key) {
        let waiter = match registry.pop_front(db_index, key) {
            Some(w) => w,
            None => return false,
        };
        let wait_id = waiter.wait_id;

        let result = match &waiter.cmd {
            BlockedCommand::XRead { streams, count } => {
                // Find this key's last_seen_id in the streams list
                let last_seen = streams.iter()
                    .find(|(k, _)| k == key)
                    .map(|(_, id)| *id);
                if let Some(last_id) = last_seen {
                    if let Ok(Some(stream)) = db.get_stream(key) {
                        let start = StreamId { ms: last_id.ms, seq: last_id.seq.saturating_add(1) };
                        let entries = stream.range(start, StreamId::MAX, *count);
                        if !entries.is_empty() {
                            // Format as [[key, [[id, [f,v,...]], ...]], ...]
                            // For single-key wakeup, just return this key's entries
                            let entry_frames: Vec<Frame> = entries.iter()
                                .map(|(id, fields)| format_entry(*id, fields))
                                .collect();
                            Some(Frame::Array(vec![
                                Frame::Array(vec![
                                    Frame::BulkString(key.clone()),
                                    Frame::Array(entry_frames),
                                ])
                            ]))
                        } else { None }
                    } else { None }
                } else { None }
            }
            BlockedCommand::XReadGroup { group, consumer, count, noack, .. } => {
                if let Ok(Some(stream)) = db.get_stream_mut(key) {
                    match stream.read_group_new(group, consumer, *count, *noack) {
                        Ok(entries) if !entries.is_empty() => {
                            let entry_frames: Vec<Frame> = entries.iter()
                                .map(|(id, fields)| format_entry(*id, fields))
                                .collect();
                            Some(Frame::Array(vec![
                                Frame::Array(vec![
                                    Frame::BulkString(key.clone()),
                                    Frame::Array(entry_frames),
                                ])
                            ]))
                        }
                        _ => None,
                    }
                } else { None }
            }
            _ => None, // List/zset commands don't watch stream keys
        };

        registry.remove_wait(wait_id);

        if let Some(frame) = result {
            let _ = waiter.reply_tx.send(Some(frame));
            return true;
        }
        let _ = waiter.reply_tx.send(None);
    }
    false
}
```
