```rust
/// Async variant of finalize: uses tokio::fs for non-blocking I/O.
///
/// Under Monoio, falls back to synchronous write (thread-per-core model,
/// rare operation, acceptable blocking).
pub async fn finalize_async(&mut self) -> anyhow::Result<()> {
    // Write EOF marker
    self.output_buf.push(EOF_MARKER);

    // Global CRC32 of entire output_buf
    let mut hasher = Hasher::new();
    hasher.update(&self.output_buf);
    let global_crc = hasher.finalize();
    self.output_buf.extend_from_slice(&global_crc.to_le_bytes());

    // Atomic write: write to .tmp, then rename
    let tmp_path = self.file_path.with_extension("rrdshard.tmp");

    // Take ownership of buffer to avoid borrow across await
    let buf = std::mem::take(&mut self.output_buf);
    let file_path = self.file_path.clone();

    #[cfg(feature = "runtime-tokio")]
    {
        tokio::fs::write(&tmp_path, &buf).await
            .context("Failed to write temporary snapshot file")?;
        tokio::fs::rename(&tmp_path, &file_path).await
            .context("Failed to rename temporary snapshot file")?;
    }

    #[cfg(feature = "runtime-monoio")]
    {
        std::fs::write(&tmp_path, &buf)
            .context("Failed to write temporary snapshot file")?;
        std::fs::rename(&tmp_path, &file_path)
            .context("Failed to rename temporary snapshot file")?;
    }

    Ok(())
}
```

**1. Add SAVE and LASTSAVE handlers to `src/command/persistence.rs`:**

```rust
/// SAVE command: synchronous save to disk. Blocks until complete.
///
/// In sharded mode, this collects data from all shards via snapshot
/// and writes a single RDB file. In single-threaded mode, saves directly.
pub fn handle_save(
    db: &SharedDatabases,
    dir: &str,
    dbfilename: &str,
) -> Frame {
    if SAVE_IN_PROGRESS.load(Ordering::SeqCst) {
        return Frame::Error(Bytes::from_static(
            b"ERR Background save already in progress",
        ));
    }

    let databases: Vec<Database> = db.iter().map(|lock| {
        let guard = lock.read();
        // Clone database for save -- this is the blocking SAVE path
        guard.clone()
    }).collect();

    let path = PathBuf::from(dir).join(dbfilename);
    match crate::persistence::rdb::save(&databases, &path) {
        Ok(()) => {
            LAST_SAVE_TIME.store(
                std::time::SystemTime::now()
                    .duration_since(std::time::UNIX_EPOCH)
                    .unwrap_or_default()
                    .as_secs(),
                Ordering::Relaxed,
            );
            Frame::SimpleString(Bytes::from_static(b"OK"))
        }
        Err(e) => Frame::Error(Bytes::from(format!("ERR {}", e))),
    }
}

/// LASTSAVE command: returns Unix timestamp of last successful save.
pub fn handle_lastsave() -> Frame {
    let ts = LAST_SAVE_TIME.load(Ordering::Relaxed);
    Frame::Integer(ts as i64)
}
```

NOTE: The `handle_save` function takes `SharedDatabases` (the Arc<Vec<RwLock<Database>>>). If the single-threaded BGSAVE path already has access to this, SAVE can reuse the same pattern. For sharded mode where there's no shared DB access, SAVE should return an error: `"ERR SAVE not supported in sharded mode, use BGSAVE"`.
