4. **EXEC handler:**

5. **DISCARD handler:**

   if cmd.as_slice() == b"DISCARD" {

       } else {

6. **WATCH handler** (must be OUTSIDE multi -- WATCH inside MULTI is an error):
   ```rust
   if cmd.as_slice() == b"WATCH" {
       if in_multi {
           framed.send(Frame::Error(Bytes::from_static(b"ERR WATCH inside MULTI is not allowed"))).await;
       } else if cmd_args.is_empty() {
           framed.send(Frame::Error(Bytes::from_static(b"ERR wrong number of arguments for 'watch' command"))).await;
       } else {
           let dbs = db.lock().unwrap();
           for arg in cmd_args {
               if let Frame::BulkString(key) = arg {
                   let version = dbs[selected_db].get_version(key);
                   watched_keys.insert(key.clone(), version);
               }
           }
           drop(dbs);
           framed.send(Frame::SimpleString(Bytes::from_static(b"OK"))).await;
       }
       continue;
   }
   ```

7. **UNWATCH handler:**
   ```rust
   if cmd.as_slice() == b"UNWATCH" {
       watched_keys.clear();
       framed.send(Frame::SimpleString(Bytes::from_static(b"OK"))).await;
       continue;
   }
   ```

8. **Command queuing** (after all intercepts, before normal dispatch):
   ```rust
   if in_multi {
       // Queue the command (re-assemble the Frame::Array since we destructured it)
       command_queue.push(frame); // Note: need to keep `frame` available here
       framed.send(Frame::SimpleString(Bytes::from_static(b"QUEUED"))).await;
       continue;
   }
   ```
   IMPORTANT: The current code extracts command info via `extract_command(&frame)` which borrows frame. For queuing, we need the original frame. Restructure: do the intercept checks first using extract_command (which borrows), then if none match and in_multi, push the original frame to queue. The key insight: extract_command borrows frame immutably, so after the borrow ends, frame can be moved into the queue.

9. **execute_transaction function** (new function in connection.rs):
   ```rust
   async fn execute_transaction(
       db: &Arc<Mutex<Vec<Database>>>,
       command_queue: &[Frame],
       watched_keys: &HashMap<Bytes, u64>,
       selected_db: &mut usize,
       aof_tx: &Option<mpsc::Sender<AofMessage>>,
       change_counter: &Option<Arc<AtomicU64>>,
   ) -> Frame {
       let mut dbs = db.lock().unwrap();
       let db_count = dbs.len();

       // Check WATCH versions -- if any key's version changed, abort
       for (key, watched_version) in watched_keys {
           let current_version = dbs[*selected_db].get_version(key);
           if current_version != *watched_version {
               return Frame::Null; // Transaction aborted
           }
       }

       // Execute all queued commands atomically (under the same lock)
       let mut results = Vec::with_capacity(command_queue.len());
       for cmd_frame in command_queue {
           let frame_clone = cmd_frame.clone();

           // Check if write for AOF
           let is_write = extract_command(&frame_clone)
               .map(|(ref cmd, _)| aof::is_write_command(cmd))
               .unwrap_or(false);

           let aof_bytes = if is_write && aof_tx.is_some() {
               let mut buf = BytesMut::new();
               crate::protocol::serialize::serialize(&frame_clone, &mut buf);
               Some(buf.freeze())
           } else {
               None
           };

           let result = dispatch(&mut dbs[*selected_db], frame_clone, selected_db, db_count);
           let response = match result {
               DispatchResult::Response(f) => f,
               DispatchResult::Quit(f) => f, // QUIT inside MULTI just returns OK, doesn't quit mid-transaction
           };

           // AOF logging for successful writes
           if let Some(bytes) = aof_bytes {
               if !matches!(&response, Frame::Error(_)) {
                   // AOF send must happen outside lock in real async code
                   // For now, just track it -- we'll send after releasing lock
                   // Actually, we need to collect and send after. Store for later.
               }
           }

           results.push(response);
       }

       Frame::Array(results)
   }
   ```

   REFINEMENT on AOF: Since execute_transaction holds the db lock (synchronous Mutex), we cannot do async sends inside it. Collect AOF entries and send them after dropping the lock:
   ```rust
   // Collect aof_entries: Vec<Bytes> during the loop
   // After drop(dbs):
   // for bytes in aof_entries { aof_tx.send(AofMessage::Append(bytes)).await; change_counter.fetch_add(1); }
   ```
   Make execute_transaction return `(Frame, Vec<Bytes>)` -- the Frame result and AOF entries to send. The caller sends AOF entries after getting the result. Or simpler: make it not async, return the result and AOF bytes, caller handles async sends.

   Actual implementation: `fn execute_transaction(...) -> (Frame, Vec<Bytes>)` (not async). Returns the result frame and a vec of AOF byte entries. The caller in the connection loop then sends AOF entries and increments change counter.
