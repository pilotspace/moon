pub fn object(db: &mut Database, args: &[Frame]) -> Frame {
    if args.is_empty() {
        return err_wrong_args("OBJECT");
    }
    let subcommand = match extract_bytes(&args[0]) {
        Some(s) => s.to_ascii_uppercase(),
        None => return err_wrong_args("OBJECT"),
    };
    match subcommand.as_ref() {
        b"ENCODING" => {
            if args.len() != 2 {
                return err_wrong_args("OBJECT");
            }
            let key = match extract_bytes(&args[1]) {
                Some(k) => k.as_ref(),
                None => return err_wrong_args("OBJECT"),
            };
            match db.get(key) {
                Some(entry) => {
                    let rv = entry.value.to_redis_value();
                    let encoding = rv.encoding_name();
                    Frame::BulkString(Bytes::from(encoding))
                }
                None => Frame::Null,
            }
        }
        b"HELP" => {
            Frame::Array(vec![
                Frame::BulkString(Bytes::from_static(b"OBJECT ENCODING <key>")),
            ])
        }
        _ => Frame::Error(Bytes::from_static(b"ERR unknown OBJECT subcommand")),
    }
}
