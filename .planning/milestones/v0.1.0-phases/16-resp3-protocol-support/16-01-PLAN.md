pub enum Frame {

    Error(Bytes),

impl PartialEq for Frame {
    fn eq(&self, other: &Self) -> bool {
        match (self, other) {
            (Self::SimpleString(a), Self::SimpleString(b)) => a == b,
            (Self::Error(a), Self::Error(b)) => a == b,
            (Self::Integer(a), Self::Integer(b)) => a == b,
            (Self::BulkString(a), Self::BulkString(b)) => a == b,
            (Self::Array(a), Self::Array(b)) => a == b,
            (Self::Null, Self::Null) => true,
            (Self::Map(a), Self::Map(b)) => a == b,
            (Self::Set(a), Self::Set(b)) => a == b,
            (Self::Double(a), Self::Double(b)) => OrderedFloat(*a) == OrderedFloat(*b),
            (Self::Boolean(a), Self::Boolean(b)) => a == b,
            (Self::VerbatimString { encoding: ae, data: ad },
             Self::VerbatimString { encoding: be, data: bd }) => ae == be && ad == bd,
            (Self::BigNumber(a), Self::BigNumber(b)) => a == b,
            (Self::Push(a), Self::Push(b)) => a == b,
            _ => false,
        }
    }
}

**Step 3: Add RESP3 match arms in `parse_single_frame`** (after the existing `b'*'` arm, before the `byte => Err(...)` catch-all):

- `b'_'` (Null): Read CRLF, return `Frame::Null`. No payload.
- `b'#'` (Boolean): Read single byte after `#`. `t` -> true, `f` -> false. Read CRLF.
- `b','` (Double): Read line until CRLF. Handle special values: `inf`, `-inf`, `nan` (case-insensitive for nan). Otherwise `str::parse::<f64>()`.
- `b'('` (BigNumber): Read line until CRLF, return `Frame::BigNumber(Bytes::copy_from_slice(line))`.
- `b'='` (VerbatimString): Read length as decimal (like BulkString), then read `length` bytes. First 3 bytes are encoding, byte 4 is `:`, rest is data. Return `Frame::VerbatimString { encoding: Bytes::copy_from_slice(&data[..3]), data: Bytes::copy_from_slice(&data[4..]) }`. Validate length >= 4 (encoding + colon).
- `b'%'` (Map): Read count as decimal. For each entry, parse TWO frames (key + value). Return `Frame::Map(pairs)`. Use `config.max_array_length` for count limit. Increment depth.
- `b'~'` (Set): Read count as decimal. Parse `count` frames. Return `Frame::Set(items)`. Same limits as Array.
- `b'>'` (Push): Read count as decimal. Parse `count` frames. Return `Frame::Push(items)`. Same limits as Array.

**Step 1: Add `serialize_resp3` function in `src/protocol/serialize.rs`:**

Add a new public function `serialize_resp3(frame: &Frame, buf: &mut BytesMut)` that writes RESP3 wire format:

```rust
pub fn serialize_resp3(frame: &Frame, buf: &mut BytesMut) {
    match frame {
        // RESP2 types serialize identically in RESP3 (except Null)
        Frame::SimpleString(s) => { buf.put_u8(b'+'); buf.put_slice(s); buf.put_slice(b"\r\n"); }
        Frame::Error(s) => { buf.put_u8(b'-'); buf.put_slice(s); buf.put_slice(b"\r\n"); }
        Frame::Integer(n) => { buf.put_u8(b':'); buf.put_slice(itoa::Buffer::new().format(*n).as_bytes()); buf.put_slice(b"\r\n"); }
        Frame::BulkString(data) => { /* same as RESP2 */ }
        Frame::Array(items) => { buf.put_u8(b'*'); /* count + recurse with serialize_resp3 */ }
        // RESP3 Null uses `_\r\n` instead of `$-1\r\n`
        Frame::Null => { buf.put_slice(b"_\r\n"); }
        // New RESP3 types
        Frame::Boolean(b) => { buf.put_u8(b'#'); buf.put_u8(if *b { b't' } else { b'f' }); buf.put_slice(b"\r\n"); }
        Frame::Double(f) => {
            buf.put_u8(b',');
            if f.is_infinite() {
                if f.is_sign_positive() { buf.put_slice(b"inf"); }
                else { buf.put_slice(b"-inf"); }
            } else if f.is_nan() {
                buf.put_slice(b"nan");
            } else {
                // Use ryu or format! for float formatting
                let s = format!("{}", f);  // stdlib is adequate
                buf.put_slice(s.as_bytes());
            }
            buf.put_slice(b"\r\n");
        }
        Frame::BigNumber(n) => { buf.put_u8(b'('); buf.put_slice(n); buf.put_slice(b"\r\n"); }
        Frame::VerbatimString { encoding, data } => {
            buf.put_u8(b'=');
            let total_len = encoding.len() + 1 + data.len(); // encoding + ':' + data
            buf.put_slice(itoa::Buffer::new().format(total_len).as_bytes());
            buf.put_slice(b"\r\n");
            buf.put_slice(encoding);
            buf.put_u8(b':');
            buf.put_slice(data);
            buf.put_slice(b"\r\n");
        }
        Frame::Map(entries) => {
            buf.put_u8(b'%');
            buf.put_slice(itoa::Buffer::new().format(entries.len()).as_bytes());
            buf.put_slice(b"\r\n");
            for (key, value) in entries {
                serialize_resp3(key, buf);
                serialize_resp3(value, buf);
            }
        }
        Frame::Set(items) => {
            buf.put_u8(b'~');
            buf.put_slice(itoa::Buffer::new().format(items.len()).as_bytes());
            buf.put_slice(b"\r\n");
            for item in items { serialize_resp3(item, buf); }
        }
        Frame::Push(items) => {
            buf.put_u8(b'>');
            buf.put_slice(itoa::Buffer::new().format(items.len()).as_bytes());
            buf.put_slice(b"\r\n");
            for item in items { serialize_resp3(item, buf); }
        }
    }
}
```
