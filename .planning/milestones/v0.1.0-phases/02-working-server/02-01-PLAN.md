1. **src/server/codec.rs** -- Implement RespCodec wrapping Phase 1 parser/serializer:
   ```rust
   use bytes::BytesMut;
   use tokio_util::codec::{Decoder, Encoder};
   use crate::protocol::{self, Frame, ParseConfig, ParseError};

   pub struct RespCodec {
       config: ParseConfig,
   }

   impl RespCodec {
       pub fn new(config: ParseConfig) -> Self {
           Self { config }
       }
   }

   impl Default for RespCodec {
       fn default() -> Self {
           Self::new(ParseConfig::default())
       }
   }

   impl Decoder for RespCodec {
       type Item = Frame;
       type Error = std::io::Error;

       fn decode(&mut self, src: &mut BytesMut) -> Result<Option<Frame>, Self::Error> {
           match protocol::parse(src, &self.config) {
               Ok(frame) => Ok(frame),
               Err(ParseError::Incomplete) => Ok(None),
               Err(e) => Err(std::io::Error::new(std::io::ErrorKind::InvalidData, e)),
           }
       }
   }

   impl Encoder<Frame> for RespCodec {
       type Error = std::io::Error;

       fn encode(&mut self, item: Frame, dst: &mut BytesMut) -> Result<(), Self::Error> {
           protocol::serialize(&item, dst);
           Ok(())
       }
   }
   ```

3. **src/server/connection.rs** -- Connection handler:
   ```rust
   use bytes::Bytes;
   use futures_util::{SinkExt, StreamExt};
   use std::sync::{Arc, Mutex};
   use tokio::net::TcpStream;
   use tokio_util::codec::Framed;
   use tokio_util::sync::CancellationToken;
   use crate::protocol::Frame;
   use crate::storage::Database;
   use super::codec::RespCodec;

   pub async fn handle_connection(
       stream: TcpStream,
       db: Arc<Mutex<Vec<Database>>>,
       shutdown: CancellationToken,
   ) {
       let mut framed = Framed::new(stream, RespCodec::default());
       let mut selected_db: usize = 0;

       loop {
           tokio::select! {
               result = framed.next() => {
                   match result {
                       Some(Ok(frame)) => {
                           let response = {
                               let mut dbs = db.lock().unwrap();
                               crate::command::dispatch(&mut dbs[selected_db], frame, &mut selected_db, dbs.len())
                           };
                           // Check if QUIT was issued (response is +OK and we should close)
                           let is_quit = matches!(&response, Frame::SimpleString(s) if s.as_ref() == b"OK_QUIT");
                           let send_frame = if is_quit {
                               Frame::SimpleString(Bytes::from_static(b"OK"))
                           } else {
                               response
                           };
                           if framed.send(send_frame).await.is_err() {
                               break;
                           }
                           if is_quit {
                               break;
                           }
                       }
                       Some(Err(_)) => break,
                       None => break,
                   }
               }
               _ = shutdown.cancelled() => {
                   let _ = framed.send(Frame::Error(Bytes::from_static(b"ERR server shutting down"))).await;
                   break;
               }
           }
       }
   }
   ```
   Note on QUIT: The dispatch function returns a special sentinel `Frame::SimpleString(Bytes::from_static(b"OK_QUIT"))` for QUIT. The connection handler detects this, sends +OK, then closes. Alternatively, dispatch can return an enum `CommandResult { Response(Frame), Quit(Frame) }` -- use your judgment on which is cleaner. The key is that QUIT sends OK then closes the connection.

   IMPORTANT: `futures_util` is NOT in Cargo.toml. The `StreamExt` and `SinkExt` traits come from `tokio_stream::StreamExt` and `futures_util::SinkExt`. Check what tokio-util re-exports. If needed, add `futures = "0.3"` to Cargo.toml or use `tokio_util::codec::Framed` methods directly. The `Framed` type implements `Stream` and `Sink` but you need the extension traits. Use `tokio_stream::StreamExt` (from tokio-stream, which tokio-util depends on) and `tokio_util::codec::Framed` which has `.send()` via `SinkExt`. Add `futures = "0.3"` if needed for `SinkExt` -- it's a lightweight dep.

4. **src/server/listener.rs** -- TCP accept loop:
   ```rust
   use std::sync::{Arc, Mutex};
   use tokio::net::TcpListener;
   use tokio_util::sync::CancellationToken;
   use tracing::{info, error};
   use crate::config::ServerConfig;
   use crate::storage::Database;
   use super::connection;

   pub async fn run(config: ServerConfig) -> anyhow::Result<()> {
       let addr = format!("{}:{}", config.bind, config.port);
       let listener = TcpListener::bind(&addr).await?;
       info!("Listening on {}", addr);

       let databases: Vec<Database> = (0..config.databases).map(|_| Database::new()).collect();
       let db = Arc::new(Mutex::new(databases));

       let token = CancellationToken::new();
       let shutdown_token = token.clone();

       tokio::spawn(async move {
           tokio::signal::ctrl_c().await.ok();
           info!("Shutdown signal received");
           shutdown_token.cancel();
       });

       loop {
           tokio::select! {
               result = listener.accept() => {
                   match result {
                       Ok((stream, addr)) => {
                           info!("New connection from {}", addr);
                           let db = db.clone();
                           let conn_token = token.child_token();
                           tokio::spawn(connection::handle_connection(stream, db, conn_token));
                       }
                       Err(e) => {
                           error!("Accept error: {}", e);
                       }
                   }
               }
               _ = token.cancelled() => {
                   info!("Server shutting down");
                   break;
               }
           }
       }
       Ok(())
   }
   ```

2. **src/command/mod.rs** -- Replace the stub dispatch with the real implementation:
   ```rust
   pub mod connection;

   use bytes::Bytes;
   use crate::protocol::Frame;
   use crate::storage::Database;

   /// Result of command dispatch. Quit variant signals connection should close after sending response.
   pub enum DispatchResult {
       Response(Frame),
       Quit(Frame),
   }

   pub fn dispatch(db: &mut Database, frame: Frame, selected_db: &mut usize, db_count: usize) -> DispatchResult {
       let args = match frame {
           Frame::Array(args) if !args.is_empty() => args,
           _ => return DispatchResult::Response(Frame::Error(
               Bytes::from_static(b"ERR invalid command format")
           )),
       };

       let cmd_name = match &args[0] {
           Frame::BulkString(s) => s.to_ascii_uppercase(),

               Bytes::from_static(b"ERR invalid command name")
           )),
       };

           b"PING" => connection::ping(cmd_args),
           b"ECHO" => connection::echo(cmd_args),
           b"QUIT" => return DispatchResult::Quit(Frame::SimpleString(Bytes::from_static(b"OK"))),
           b"SELECT" => connection::select(cmd_args, selected_db, db_count),
           b"COMMAND" => connection::command(cmd_args),
           b"INFO" => connection::info(db, cmd_args),
           _ => Frame::Error(Bytes::from(format!(
               "ERR unknown command '{}', with args beginning with: ",
               String::from_utf8_lossy(&cmd_name)
           ))),
       };

       DispatchResult::Response(response)
   }
   ```
