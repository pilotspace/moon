   For `run_sharded` (the sharded path is what main.rs uses):
   ```rust
   #[cfg(feature = "runtime-monoio")]
   pub async fn run_sharded(
       config: ServerConfig,
       conn_txs: Vec<channel::MpscSender<crate::runtime::TcpStream>>,
       shutdown: CancellationToken,
   ) -> anyhow::Result<()> {
       let addr = format!("{}:{}", config.bind, config.port);
       let listener = monoio::net::TcpListener::bind(&addr)?;
       let num_shards = conn_txs.len();
       info!("Listening on {} ({} shards, monoio)", addr, num_shards);
       let mut next_shard: usize = 0;

       // Ctrl+C handler -- ctrlc crate sets handler on OS thread, signals our token
       let shutdown_signal = shutdown.clone();
       ctrlc::set_handler(move || {
           info!("Shutdown signal received");
           shutdown_signal.cancel();
       }).expect("failed to set Ctrl+C handler");

       loop {
           monoio::select! {
               result = listener.accept() => {
                   match result {
                       Ok((stream, addr)) => {
                           debug!("New connection from {} -> shard {}", addr, next_shard);
                           // Convert monoio TcpStream to std::net::TcpStream for cross-thread send
                           // monoio TcpStream has into_raw_fd(); use that to create std TcpStream
                           let std_stream = unsafe {
                               use std::os::unix::io::IntoRawFd;
                               use std::os::unix::io::FromRawFd;
                               let fd = stream.into_raw_fd();
                               std::net::TcpStream::from_raw_fd(fd)
                           };
                           let tx = &conn_txs[next_shard];
                           if tx.send(std_stream).is_err() {
                               error!("Failed to send connection to shard {}", next_shard);
                           }
                           next_shard = (next_shard + 1) % num_shards;
                       }
                       Err(e) => {
                           error!("Accept error: {}", e);
                       }
                   }
               }
               _ = shutdown.cancelled() => {
                   info!("Listener shutting down");
                   break;
               }
           }
       }
       Ok(())
   }
   ```

   Note: Use `send` (blocking) not `send_async` since flume's blocking send works fine in this context and avoids needing the monoio runtime to drive the channel future. Actually, use `tx.send_async(std_stream).await` since we're in an async context within monoio -- flume's `send_async` returns a future that any runtime can poll.

   For `run()` (non-sharded path): Add a minimal `#[cfg(feature = "runtime-monoio")]` version that returns `anyhow::bail!("Non-sharded mode not supported under monoio -- use --shards 1")`. The sharded path with 1 shard is the MVP.
