Monoio read_exact equivalent (no native read_exact — must loop):
```rust
async fn monoio_read_exact(stream: &mut monoio::net::TcpStream, total: usize) -> std::io::Result<Vec<u8>> {
    use monoio::io::AsyncReadRent;
    let mut result = Vec::with_capacity(total);
    while result.len() < total {
        let remaining = total - result.len();
        let buf = vec![0u8; remaining];
        let (res, buf) = stream.read(buf).await;
        let n = res?;
        if n == 0 { return Err(std::io::Error::new(std::io::ErrorKind::UnexpectedEof, "EOF")); }
        result.extend_from_slice(&buf[..n]);
    }
    Ok(result)
}
```
